\chapter{Standard library}
\label{sec:20-stdlib}
\index{standard library}

\section{Overview}
\label{sec:stdlib.overview}

\pnum
The Lain standard library is a collection of modules under the \lain{std}
namespace (\S\,\ref{sec:mod.stdlib}).  All standard library modules are
written in Lain, using C interop (\S\,\ref{sec:17-interop}) for
system-level operations.  They are subject to the same ownership,
borrow-checking, and purity rules as user code.

\pnum
This clause is normative for the modules listed in Table~\ref{tab:stdlib-index}.
All other modules described in informative documentation are not part of
this specification.

\begin{table}[htbp]
\caption{Standard library module index}
\label{tab:stdlib-index}
\begin{tabular}{@{}lll@{}}
\toprule
\textbf{Module path} & \textbf{Description} & \textbf{Status} \\
\midrule
\lain{std.c}    & C standard library bindings    & Normative \\
\lain{std.io}   & Console I/O                    & Normative \\
\lain{std.fs}   & File system operations         & Normative \\
\lain{std.math} & Pure mathematical functions    & Normative \\
\bottomrule
\end{tabular}
\end{table}

\begin{constraint}
  A conforming implementation shall provide all modules listed in
  Table~\ref{tab:stdlib-index} at the specified module paths.
\end{constraint}

\section{\texttt{std.c} — C standard library bindings}
\label{sec:stdlib.c}
\index{std.c}

\subsection{Purpose}
\label{sec:stdlib.c.purpose}

\pnum
\lain{std.c} provides low-level bindings to selected C standard library
functions.  It is the foundation on which all other standard library
modules that require system calls are built.

\subsection{Required C headers}
\label{sec:stdlib.c.headers}

\pnum
\lain{std.c} emits the following C include directives into the translation
output:

\begin{laincode}
c_include "<stdio.h>"
c_include "<stdlib.h>"
\end{laincode}

\subsection{Opaque types}
\label{sec:stdlib.c.types}

\pnum
\lain{std.c} declares the following opaque C type:

\begin{laincode}
extern type FILE
\end{laincode}

\lain{FILE} is an opaque type representing a C standard I/O stream.  It
may only be used through pointers and passed to the \lain{extern}
functions declared below.

\begin{constraint}
  \lain{FILE} shall not be used as a value type.  A declaration such as
  \lain{var f FILE} is \illformed{} (\S\,\ref{sec:interop.opaque}).
\end{constraint}

\subsection{Extern declarations}
\label{sec:stdlib.c.extern}

\pnum
The following C functions are declared in \lain{std.c}:

\begin{longtable}{@{}ll@{}}
\toprule
\textbf{Declaration} & \textbf{Ownership notes} \\
\midrule
\endfirsthead
\toprule
\textbf{Declaration} & \textbf{Ownership notes} \\
\midrule
\endhead
\bottomrule
\endlastfoot
\lain{extern proc libc\_printf(fmt *u8, ...) int}
  & Shared \lain{fmt}; variadic \\
\lain{extern proc libc\_puts(s *u8) int}
  & Shared \lain{s} \\
\lain{extern proc fopen(filename *u8, mode *u8) mov *FILE}
  & Returns owned handle \\
\lain{extern proc fclose(stream mov *FILE) int}
  & Consumes handle \\
\lain{extern proc fputs(s *u8, stream *FILE) int}
  & Shared args \\
\lain{extern proc fgets(s var *u8, n int, stream *FILE) var *u8}
  & Mutable buffer \\
\end{longtable}

\begin{note}
  \lain{libc\_printf} and \lain{libc\_puts} are renamed to avoid
  conflicts with user-defined functions named \lain{printf} and
  \lain{puts}.
\end{note}

\section{\texttt{std.io} — Console I/O}
\label{sec:stdlib.io}
\index{std.io}

\subsection{Dependencies}
\label{sec:stdlib.io.deps}

\pnum
\lain{std.io} imports \lain{std.c}.

\subsection{\texttt{print}}
\label{sec:stdlib.io.print}
\index{print}

\begin{functionspec}{print}{std.io}
\synopsis{proc print(s [u8])}
\noindent\textit{Effect:} Writes the byte sequence
\texttt{s.data[0..s.len-1]} to standard output.  No trailing newline
is appended.\\
\noindent\textit{Parameter \texttt{s}:} Shared borrow of a byte slice.
\end{functionspec}

\subsection{\texttt{println}}
\label{sec:stdlib.io.println}
\index{println}

\begin{functionspec}{println}{std.io}
\synopsis{proc println(s [u8])}
\noindent\textit{Effect:} Writes the byte sequence
\texttt{s.data[0..s.len-1]} to standard output, followed by a newline
character (\texttt{0x0A}).\\
\noindent\textit{Parameter \texttt{s}:} Shared borrow of a byte slice.
\end{functionspec}

\section{\texttt{std.fs} — File system}
\label{sec:stdlib.fs}
\index{std.fs}

\subsection{Dependencies}
\label{sec:stdlib.fs.deps}

\pnum
\lain{std.fs} imports \lain{std.c}.

\subsection{Type: \texttt{File}}
\label{sec:stdlib.fs.file}
\index{File type}

\pnum
\lain{File} is a linear struct that wraps a C \texttt{FILE*} handle:

\begin{laincode}
type File {
    mov handle *FILE
}
\end{laincode}

\pnum
Because \lain{handle} is annotated \lain{mov}, \lain{File} is a linear
type.  It must be consumed exactly once, typically by calling
\lain{close\_file}.

\subsection{\texttt{open\_file}}
\label{sec:stdlib.fs.open}
\index{open\_file}

\begin{functionspec}{open\_file}{std.fs}
\synopsis{proc open\_file(path [u8], mode [u8]) mov File}
\noindent\textit{Effect:} Opens the file at \texttt{path} with the access
mode \texttt{mode} (C \texttt{fopen} semantics).  Returns an owned
\lain{File} value.  The caller is responsible for closing the file by
consuming the \lain{File}.\\
\noindent\textit{Parameter \texttt{path}:} Shared borrow of a
null-terminated path string.\\
\noindent\textit{Parameter \texttt{mode}:} Shared borrow of a mode string
(\texttt{"r"}, \texttt{"w"}, \texttt{"a"}, etc.).
\end{functionspec}

\begin{note}
  If \texttt{fopen} returns null, the \lain{handle} field is a null
  pointer.  Subsequent operations on such a file produce C-level undefined
  behavior.  A future version may return \lain{Result(File, int)} instead.
\end{note}

\subsection{\texttt{close\_file}}
\label{sec:stdlib.fs.close}
\index{close\_file}

\begin{functionspec}{close\_file}{std.fs}
\synopsis{proc close\_file(mov \{handle\} File)}
\noindent\textit{Effect:} Closes the file by calling C \texttt{fclose} on
the extracted \texttt{handle} pointer.  Consumes the \lain{File} value.\\
\noindent\textit{Parameter:} The \lain{File} value is consumed by the call;
it shall not be used after this point.
\end{functionspec}

\subsection{\texttt{write\_file}}
\label{sec:stdlib.fs.write}
\index{write\_file}

\begin{functionspec}{write\_file}{std.fs}
\synopsis{proc write\_file(f File, s [u8])}
\noindent\textit{Effect:} Writes the byte sequence \texttt{s} to the file
\texttt{f} using C \texttt{fputs}.\\
\noindent\textit{Parameter \texttt{f}:} Shared borrow of the file handle.\\
\noindent\textit{Parameter \texttt{s}:} Shared borrow of the string to write.
\end{functionspec}

\begin{example}
\begin{laincode}
import std.fs

proc main() int {
    var f = open_file("log.txt", "w")
    defer { close_file(mov f) }
    write_file(f, "Hello, world!\n")
    return 0
}
\end{laincode}
\end{example}

\section{\texttt{std.math} — Mathematical functions}
\label{sec:stdlib.math}
\index{std.math}

\subsection{Overview}
\label{sec:stdlib.math.overview}

\pnum
All functions in \lain{std.math} are declared \lain{func} (pure and
terminating).  They may be called from any context, including other
\lain{func} bodies.  \lain{std.math} has no dependencies.

\subsection{\texttt{min}}
\label{sec:stdlib.math.min}
\index{min}

\begin{functionspec}{min}{std.math}
\synopsis{func min(a int, b int) int}
\noindent\textit{Returns:} The smaller of \texttt{a} and \texttt{b}.
If \texttt{a == b}, either may be returned (they are equal).
\end{functionspec}

\subsection{\texttt{max}}
\label{sec:stdlib.math.max}
\index{max}

\begin{functionspec}{max}{std.math}
\synopsis{func max(a int, b int) int}
\noindent\textit{Returns:} The larger of \texttt{a} and \texttt{b}.
\end{functionspec}

\subsection{\texttt{abs}}
\label{sec:stdlib.math.abs}
\index{abs}

\begin{functionspec}{abs}{std.math}
\synopsis{func abs(x int) int}
\noindent\textit{Returns:} The absolute value of \texttt{x},
i.e., \texttt{x} if \texttt{x >= 0}, \texttt{-x} otherwise.
\end{functionspec}

\begin{note}
  \lain{abs(INT\_MIN)} produces implementation-defined behaviour because
  \texttt{|INT\_MIN|} is not representable as \lain{int} on two's
  complement platforms.
\end{note}

\subsection{\texttt{clamp}}
\label{sec:stdlib.math.clamp}
\index{clamp}

\begin{functionspec}{clamp}{std.math}
\synopsis{func clamp(x int, lo int, hi int) int}
\noindent\textit{Returns:} \texttt{x} clamped to the inclusive range
\texttt{[lo, hi]}, i.e.,
\texttt{max(lo, min(x, hi))}.
\end{functionspec}

\begin{constraint}
  Behaviour is \undefbehav{undefined} if \lain{lo > hi}.
\end{constraint}

\section{Generic types (informative)}
\label{sec:stdlib.generic}

\pnum
This section describes generic type constructors that are part of the
standard library but are not normative in the current version of this
specification.

\subsection{\texttt{Option(T)}}
\label{sec:stdlib.option}
\index{Option type}

\pnum
\lain{Option(T)} is a type constructor for optional values:

\begin{laincode}
func Option(comptime T type) type {
    return type {
        Some { value T }
        None
    }
}
\end{laincode}

\pnum
Usage:

\begin{laincode}
type OptionInt = Option(int)
var x = OptionInt.Some(42)
case x {
    Some(v): libc_printf("value: %d\n", v)
    None:    libc_printf("no value\n")
}
\end{laincode}

\subsection{\texttt{Result(T, E)}}
\label{sec:stdlib.result}
\index{Result type}

\pnum
\lain{Result(T, E)} is a type constructor for fallible computations:

\begin{laincode}
func Result(comptime T type, comptime E type) type {
    return type {
        Ok  { value T }
        Err { error E }
    }
}
\end{laincode}

\pnum
Usage:

\begin{laincode}
type FileResult = Result(File, int)

proc try_open(path [u8]) FileResult {
    var f = open_file(path, "r")
    return FileResult.Ok(mov f)
}
\end{laincode}
